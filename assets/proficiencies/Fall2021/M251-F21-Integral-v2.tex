\problem{12}  Compute the following definite/indefinite integrals.
\begin{subproblems}

\item $\ds \int_1^9 \frac{x+1}{\sqrt{x}} \; dx$
\vfill

\item $\ds \int_0^{1/2} (2-\sin(\pi x))\; dx$
\vfill

\item $\ds \int (x+1)(2x+3)\; dx$
\vfill

\newpage
\item $\ds \int \frac{e^x}{(5+e^x)^4}\; dx$
\vfill


\item $\ds \int \frac{1-\sin(x)}{x+\cos(x)}\; dx$
\vfill

\item $\ds \int  xe^{2x^2}\; dx$
\vfill

\newpage
\item $\ds \int x(x+1)^{12} \; dx$
\vfill

\item $\ds \int \sec(1-3x)\tan(1-3x) \; dx$
\vfill

\item $\ds \int \frac{8}{1+x^2} \; dx$
\vfill

\newpage

\item $\ds \int \sqrt{3}\sec^2(x)\; dx$
\vfill


\item $\ds \int \left(\sqrt[3]{x^4}+\sqrt[3]{5}\right)\; dx$
\vfill

\item $\ds \int \left(\frac{1}{x}+\frac{\ln(x)}{x}\right)\; dx$
\vfill

\end{subproblems}
