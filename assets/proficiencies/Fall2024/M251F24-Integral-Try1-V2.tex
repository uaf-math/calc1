\item There are 12 points possible on this proficiency, one point per problem. {\bf No partial credit will be given.}

\item You have one hour to complete this proficiency.

\item No aids (book, calculator, etc.) are permitted.  

\item You do \textbf{not} need to simplify your expressions.

\item Correct parenthesization is required.

\item Do not put a $+C$ where it does not belong, and you must include $+C$ where it is needed.

\problem{12}  Compute the following integrals.

\begin{subproblems}


\item   $\ds \int (x^{-3}  - e^x + 2x^5) \: dx$\\
%power rule with negative exponent, e^x, a super simple beginning problem
\vfill

\item   $\ds \int  \frac{3}{5x-1} \: dx$
%u is the denominator
\vfill

\item   $\ds \int \left(\sin \theta + \sec \theta \tan \theta  + \csc(\pi/4)\right) \: d\theta$\\
% cos, sec*tan, constant
\vfill

\newpage

\item   $\ds \int e^x\cos(e^x+1) \: dx$ \\
%SUPER EASY u-sub: sin(u),sin(x)
\vfill

\item   $\ds \int \pi\left(\frac{x-5}{2}\right) \: dx$\\
%managing simple constants
\vfill

\item   $\ds \int \frac{1+\ln(x)}{2x} \: dx$
%u-sub using deductive reasoning
\vfill

\newpage

\item   $\ds \int \left(\frac{1}{x} + e^{3x} + \sec^2(2x) \right) \: dx$ \\
%u-sub: e^u,sec^2(u),1/x
\vfill

\item   $\ds \int_0^{\pi/2} \frac{3 \sin(x)}{\sqrt{1+5\cos(x)}}\: dx$
%definite integral requiring a substitution,u-sub: 1/sqrt(u)
\vfill

\item   $\ds \int \frac{e^3}{1+x^2} \: dx$\\
% arctan
\vfill

\newpage

\item $\ds \int_1^4 \frac{x^2 - 2\sqrt{x}}{x} \: dx$\\
%quotient requiring simplification,definite integral without substitution,power rule with fractional exponent
\vfill

\item   $\ds \int bx^p \: dx$ \hspace{.1in}   where $b$ and $p$ are positive constants
%parameters
\vfill

\item   $\ds \int x(x+2)^{9}\: dx$\\
%sophisticated u-sub
\vfill

\end{subproblems}
