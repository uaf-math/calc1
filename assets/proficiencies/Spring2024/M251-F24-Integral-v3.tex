\problem{12}  Compute the following definite/indefinite integrals.
\begin{subproblems}
%basic rules: power, ln, contants
\item $\ds \int \left( \frac{x}{3}+\frac{4}{x} +\frac{4}{3}\right)\; dx$
\vfill

%easy u-sub 
%inside sine
\item $\ds \int \cos(2x)\; dx$
\vfill

%Def integral + substitution
\item $\ds \int_1^{2} x(x+1)\; dx$
\vfill


\newpage
%easy u-sub for ln
\item $\ds \int \frac{1+\sec^2(2x)}{2x+\tan(2x)}\; dx$
\vfill
%thoughtful u-sub, power rule, 
%exact homework problems
\item $\ds \int \frac{5}{x(\ln x)^3}\; dx$
\vfill
%Basic rules
\item $\ds \int (4x^3+\sin(x)+e^x) \; dx$
\vfill

\newpage
%arc sine
\item $\ds \int  \frac{1}{\sqrt{1-x^2}}\; dx$
\vfill
%sqrt in denominator, u-sub
\item $\ds \int \frac{x^3}{\sqrt{3-x^4}} \; dx$
\vfill
%no sub + u-sub
%e and trig
\item $\ds \int (e^{-x}+\sec(x)\tan(x)) \; dx$
\vfill

\newpage
%process of elimination u-sub 
%exact homrework problems
\item $\ds \int \frac{\tan^{-1}(x)}{1+x^2}\; dx$
\vfill
%definite integral with u-sub
\item $\ds \int_0^2 2x(x^2-2)^3\; dx$
\vfill
%sophisticated substitution
\item $\ds \int \frac{x}{x-1} \: dx$
\vfill

\end{subproblems}
