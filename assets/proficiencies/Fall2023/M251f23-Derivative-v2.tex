\begin{subproblems}

\vfill

%simple e^u with chain

\item   $\ds f(x)=e^{5-x^7}$
\vfill

%log and chain
%tan
%sec
\item   $\ds f(x)=\sqrt{4x-\ln(5x)}$
\vfill

\newpage
%fractions
%fractional powers
%understanding of constants
\item   $\ds f(x)=\frac{\cos(x/3)}{x^5}$
\vfill

%base other than e
%product rule inside of chain rule
\item   $\ds f(x)=\frac{1}{9x}+\sqrt{6-x}+6^9$
\vfill

%double chain
%cosine
\item   $\ds f(x) = \ln(\tan x + \sec x)$
\vfill

\newpage


\item   $\ds \tan^{-1}(x^3)$
%$\ds f(x) = x^2\sin x \ln x$
\vfill

\item   $\ds f(t) = \frac{t\ln t}{\ln 2}$
\vfill
%understanding of constants
%thoughtful algebra
%quotient/product
\item   $\ds f(x) = \log_5 (x^3)$
\vfill

\newpage
%chain inside of product
%decimal exponent
\item   $\ds f(x) = \pi \cos\left(\frac{8+x}{12}\right)$
\vfill

\item  $\ds f(x) = (\sin(x^3+e^3))^5$
\vfill

\item  Find $\ds \frac{dy}{dx}$ for $\ds 2y+2x = ye^x+1$. You must solve for $\ds \frac{dy}{dx}$.
