		\vfill

\item How to simplify \scalebox{1.5}{$\dfrac{\left(\frac{a}{b}\right)}{\left(\frac{c}{d}\right)}=$}
	\begin{enumerate}
	\item Choose integer numerical values for $a, b, c$ and $d$ to demonstrate that the rule above is correct.
	\vfill
	\item Find numerical values for $a, b, c$ and $d$ that demonstrate that to following approach is WRONG:\\
	(WRONG $\longrightarrow$) \scalebox{1.5}{$\dfrac{\left(\frac{a}{b}\right)}{\left(\frac{c}{d}\right)}=\dfrac{ac}{bd}$}
	\vfill
	\newpage
	\item Use the rule above to simplify\\
	\scalebox{1.5}{$\dfrac{\left(\frac{a}{b}\right)}{\left(c\right)}=$} \hspace{2in} and \scalebox{1.5}{$\dfrac{\left(a\right)}{\left(\frac{c}{d}\right)}=$}\hspace{1in}\\
	
	(hint: Use the fact that $r=\frac{r}{1}.$)

	\vfill
	\item (2.3 \# 99) Simplify $\frac{\cos \theta}{\cot \theta}$

	\end{enumerate}
