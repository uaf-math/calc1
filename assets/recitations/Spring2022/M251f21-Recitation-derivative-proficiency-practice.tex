\begin{itemize}
\item 
There are 12 points possible on this proficiency: one point per problem with
no partial credit. 

\item You have 30 minutes to complete this proficiency.

\item No aids (book, calculator, etc.) are permitted.  

\item You do \textbf{not} need to simplify your expressions.

\item Your final answers should start with $f'(x)=$, $dy/dx=$ or
something similar.

\end{itemize}

\problem{12}  Compute the derivatives of the following functions.
\begin{subproblems}
\item $\ds f(x) = \frac{x^{e}}{5} + 7 e^x +\sqrt{5}$
\vfill
\item $\ds f(t) = \frac{t^3-t^\frac{3}{2}+1}{t}$
\vfill
\item $\ds f(x) = (x^4-2x)\tan(x)$
\vfill
\newpage
\item $\ds f(x) = \frac{1+e^{-3x}}{\cos(3x)}$
\vfill
\item $\ds f(x) = \frac{1}{\sqrt{x}}+\left(\frac{2}{3x} \right)^{2}+\sec(x)$
\vfill
\item $\ds f(t) =\tan^{-1}(2t)+ t\ln(at+b)$ where $a$ and $b$ are a fixed constants
\vfill
\newpage
\item $\ds f(x)=\frac{\ln(x^2+1)^4}{\csc(x)}$
\vfill
\item $ \ds f(z) = \ds \sin^{-1}(\sqrt{z})$
\vfill
\item $\ds f(t) = \ds \ln(\tan(1+t^2))$
\vfill
\newpage
\item $\ds f(x) = \ds \sin^5(e^{-x} +x)$
\vfill
\item $\ds f(x) = \frac{1}{4x^2}+\left(\frac{3-x}{2}\right)^2$
\vfill
\item Compute $dy/dx$ if\quad $\ds e^y+x^2=1-xy$. You must solve for $dy/dx$.
\vfill
\end{subproblems}
